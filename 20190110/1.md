/* prog11_6, 結構陣列的大小 */
#include <stdio.h>
#include <stdlib.h>
int main(void)
{
   struct student		/* 定義結構 */
   {
       char name[10];
       int KSUID;
       int program;
   }s[1];
   int i=0;
  for(i=0;i<=1;i++)
  {
  printf("PUT 姓名:");
  gets(s[i].name);
  printf("PUT KSUID:");
  scanf("%d",&s[i].KSUID);
  printf("PUT program:");
  scanf("%d",&s[i].program);
  fflush(stdin);
  printf("\n");
  printf("\n");
  printf("\n");
  }
  
  for(i=0;i<=1;i++)
  printf("學號%d的%s同學程式設計成績是%d\n",s[i].KSUID,s[i].name,s[i].program);
  
   
   
   printf("sizeof(student[3])=%d\n",sizeof(s[3]));
   printf("sizeof(student)=%d\n",sizeof(s));
   system("pause");
   return 0;
}
